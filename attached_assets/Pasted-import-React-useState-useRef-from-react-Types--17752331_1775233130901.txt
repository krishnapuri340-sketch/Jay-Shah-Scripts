import React, { useState, useRef } from “react”;

// ─── Types ────────────────────────────────────────────────────────────────────

interface FantasyTeam {

name: string;

}

interface BanterProps {
// Match info

homeTeamCode: string;

// Algo pick (from your predictNextMatch fn)

algoReason?: string;

// Current logged-in user

}

// ─── Persona nicknames (feel free to customise) ───────────────────────────────

rajveer:  { emoji: “👑”, nickname: “Rajveer” },
mombasa:  { emoji: “🦁”, nickname: “Mombasa” },
mumbai:   { emoji: “🌊”, nickname: “Mumbai” },
ponygoat: { emoji: “🐐”, nickname: “PonyGoat” },
};

// ─── Helper ───────────────────────────────────────────────────────────────────

function getRank(teamScores: BanterProps[“teamScores”], id: string): number {
return teamScores.findIndex(s => s.id === id) + 1;
}

function ordinal(n: number) {
if (n === 1) return “1st”;
if (n === 2) return “2nd”;
if (n === 3) return “3rd”;
return `${n}th`;
}
